CMAKE_MINIMUM_REQUIRED(VERSION 3.22)
PROJECT(PRISM)

SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")

SET(MBEDTLS_DIR     ${CMAKE_CURRENT_SOURCE_DIR}/mbedtls-flat)
SET(LIBUV_DIR       ${CMAKE_CURRENT_SOURCE_DIR}/libuv)
SET(SS_DIR          ${CMAKE_CURRENT_SOURCE_DIR}/shadowsocks-crypto)
SET(PRISMUI_DIR     ${CMAKE_CURRENT_SOURCE_DIR}/prism)

# 设置头文件包含目录
SET(LUMINOUS_ADDTION_HDR
        ${LIBUV_DIR}/include
        ${MBEDTLS_DIR}/include
        )
INCLUDE_DIRECTORIES(${LUMINOUS_ADDTION_HDR})

SET(ENABLE_PROGRAMS     0)
SET(ENABLE_TESTING      0)

ADD_DEFINITIONS(-D MBEDTLS_ALLOW_PRIVATE_ACCESS)

ADD_SUBDIRECTORY(${LIBUV_DIR})
ADD_SUBDIRECTORY(${MBEDTLS_DIR})
ADD_SUBDIRECTORY(${SS_DIR})
ADD_SUBDIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}/TESTPG)
ADD_SUBDIRECTORY(${PRISMUI_DIR})
